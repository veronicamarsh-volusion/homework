using System;
					
public class Program
{
	public static void Main()
	{
		for (int i = 1; i <= 100; i++)
		{
			if (i % 3 == 0 && i % 5 == 0)
			{
				Console.WriteLine("bangpow");
			}
			else if (i % 3 == 0)
			{
				Console.WriteLine("bang");
			}
			else if (i % 5 == 0)
			{
				Console.WriteLine("pow");
			}
			else{
				Console.WriteLine(i);
	
			}
			
		}
	}
}
